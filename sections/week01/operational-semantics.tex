\section{Some things worth knowing for the Module}
Judgement form: $\types{\Gamma}{e}{\tau}$, evaluation $e \steps e'$.

\subsection{Typing rule example}
\[
\inferrule*[Right=App]
  {\types{\Gamma}{e_1}{\tau_1\!\to\!\tau_2} \quad \types{\Gamma}{e_2}{\tau_1}}
  {\types{\Gamma}{e_1\,e_2}{\tau_2}}
\]

\subsection{BNF for a tiny lambda-calculus}
\begin{grammar}
<expr> ::= <expr> <expr> \alt "\textbackslash" <var> "." <expr> \alt <var>
<var>  ::= "x" \alt "y" \alt "z"
\end{grammar}
